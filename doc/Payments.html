def get_pos_by_id(id)
  id = id.to_i
  @@pos_table.each do |k, v|
    return v if v.pos_id == id
  end
  nil
end

def init
  filename = File.join(Rails.root, 'config', 'payments.yml')
  if File.exist?(filename)
    config = YAML.load_file(filename)
    config.each do |k, v|
      pos = Pos.new(v)
      @@pos_table[k] = pos
    end
  end
end
